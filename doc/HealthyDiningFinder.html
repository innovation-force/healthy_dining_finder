def self.dishes_by_sodium(address, keyword, sodium, radius)
    #DishesBySodiumbyAddressRadiusJSON - This unique service allows you to search by location and sodium value for dish. Key here is to know that sodium numbers tend to be in the 500 plus range.
    #In addition, this service can also include an ingredient or a keyword which would appear in the name or description of the dish. For example Turkey.
    target_url = URI("#{BASE_URL}/DishesBySodiumbyAddressRadiusJSON.svc/dish/sodium/#{ TOKEN + "/" + address + "/" + keyword + "/" + sodium + "/" + radius }")
    json_response Net::HTTP.get(target_url)
end

def self.restaurant_locations_dishes_all_details(restaurant_id)
    #RestaurantLocationsDishesAllDetailsJSON - This shows all dishes available at a particular location.
    #HDF clients share the same menu across all locations but this was created in case a future restaurant has certain local meals.
    target_url = URI("#{BASE_URL}RestaurantLocationsDishesAllDetailsJSON.svc/Restaurant/#{ TOKEN + "/" + restaurant_id }")
    json_response Net::HTTP.get(target_url)
end
